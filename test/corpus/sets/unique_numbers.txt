================================================================================
Sets unique numbers
================================================================================

function main() {
    let nums = [1, 2, 1, 2, 1, 2, 3, 4, 5, 1, 2]
    mut set = {nums[0]}
    for i in 0..nums.size() {
        set.add(nums[i])
    }
    println("{}", set.size())
}
