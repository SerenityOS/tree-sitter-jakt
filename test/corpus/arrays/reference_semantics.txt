function change_value(mut vector: [String]) {
    vector[1] = "bar"
}

function main() {
    mut v = ["foo", "foo"]
    change_value(vector: v)

    mut i = 0
    while i < v.size() as! i64 {
        println("{}", v[i])
        ++i
    }
}
