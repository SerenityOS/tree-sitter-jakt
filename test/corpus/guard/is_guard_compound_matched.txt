enum Foo {
    Bar(i64)
}

function main() {
    let foo = Foo::Bar(5)
    let bar = Foo::Bar(7)

    guard foo is Bar(x) and x < 10 else {
        println("not matched")
        return -1
    }

    println("hello there: {}", x)
}
