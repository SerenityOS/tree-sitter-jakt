Structs value semantics
================================================================================

struct Foo {
    x: i64
}

function set_x(mut foo: Foo, x: i64) {
    foo.x = x
}

function main() {
    let foo = Foo(x: 5)
    println("{}", foo.x)

    set_x(foo: foo, x: 10)
    println("{}", foo.x)

    mut bar = foo
    bar.x = 15
    println("{}", foo.x)
}
