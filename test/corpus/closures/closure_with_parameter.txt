fn find(anon arr: [i64], anon cb: fn(a: i64) -> bool) -> i64? {
    for i in 0..arr.size() {
        if cb(a: arr[i]) {
            return arr[i]
        }
    }
    return None
}

    println("{}", find(arr, fn(a: i64) => a == 5))
