class Foo {
    public x: i64

    public function set(mut this, value: i64) {
        this.x = value
    }

    public function get(this) {
        return this.x
    }
}

function main() {
    mut foo = Foo(x: 1)
    mut opt_foo: Foo? = foo

    opt_foo!.set(value: 2)

    println("{}", opt_foo!.get())
}
