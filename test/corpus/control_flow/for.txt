struct Countdown {
    from: i64

    function next(mut this) -> i64? {
        if this.from == -1 {
            return None
        }

        let value = this.from
        --this.from
        return value
    }
}

function main() {
    for x in Countdown(from: 10) {
        println("{}", x)
    }
}
