function get_unique(nums: [i64]) throws -> {i64} {
    mut set = {nums[0]}
    for i in 1..nums.size() {
        set.add(nums[i])
    }
    return set
}

function main() {
    let nums = [1, 2, 3, 4, 5, 5]
    let set = get_unique(nums)
    println("{}", set.size())
}
