/// Expect:
/// - output: "foo\nfoo\n"

class Foo {
    public x: String
}

fn main() {
    let foo = Foo(x: "foo")
    let dict = ["a": foo]
    println("{}", dict["a"].x)
    println("{}", dict.get("a")!.x)
}
